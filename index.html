.. logger.js

// logger.js
const logPrefix = require('log-prefix')
const cluster = require('cluster')

module.exports = function () {
  if (!cluster.worker) {
    console.log(`> APP_NAME - UP | HOST ${global.host} | PORT ${global.config.server.appPort}`)
    console.log(`[${new Date().toISOString()}][ MASTER ] PID ${process.pid}`)
  } else {
    console.log(`[${new Date().toISOString()}][WORKER ${cluster.worker.id}] PID ${process.pid}`)
  }

  logPrefix(() => `[${new Date().toISOString()}][${(cluster.worker ? 'WORKER ' + cluster.worker.id : ' MASTER ')}]`)
}

.. sticky socket cluster

var sticky = require('socketio-sticky-session')

// config.js
const config = {
  sticky: {
    proxy: true,
    ignoreMissingHeader: true
  }
}

sticky(global.config.sticky, function () {
  var http = require('http').createServer()
  global.io = require('socket.io')(http)

  // Do whatever you want!

  return http
}).listen(global.config.server.appPort, function () {
  console.log('[STARTED]')
})
